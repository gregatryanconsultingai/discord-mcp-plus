```dockerfile
COPY package*.json ./

# Production stage
FROM node:20-alpine
WORKDIR /app
COPY package*.json ./
RUN npm ci --omit=dev
COPY --from=build /app/dist ./dist
ENV DISCORD_MCP_TRANSPORT=http
ENV DISCORD_MCP_HTTP_PORT=3000
```

### Task 3: GitHub Actions publish workflow

**Files:**
- Create: `.github/workflows/publish.yml`

- [ ] **Step 1: Create `.github/workflows/publish.yml`**

```yaml
name: Publish

on:
  push:
    tags:
      - 'v*'

jobs:
  publish-npm:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          registry-url: 'https://registry.npmjs.org'

      - run: npm ci

      - run: npm run build

      - run: npm publish --access public
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}

  publish-docker:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_TOKEN }}

      - uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            gregatryanconsultingai/discord-mcp-plus:latest
            gregatryanconsultingai/discord-mcp-plus:${{ github.ref_name }}
```

- [ ] **Step 2: Verify the YAML is valid**

```bash
npx js-yaml .github/workflows/publish.yml
```

Expected: No output (valid YAML).

- [ ] **Step 3: Commit**

